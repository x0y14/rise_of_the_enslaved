            custom_tooltip = roe_sell_slaves_through_market_decision_choose_slavic_effect_tt
            show_as_tooltip = {
                add_gold = {
                    value = roe_min_slave_token_unit_value
                    multiply = roe_slave_price_per_unit_value
                }
            }
        }
        else_if = {
            limit = { scope:choose_turkic = yes }
            custom_tooltip = roe_sell_slaves_through_market_decision_choose_turkic_effect_tt
            show_as_tooltip = {
                add_gold = {
                    value = roe_min_slave_token_unit_value
                    multiply = roe_slave_price_per_unit_value
                }
            }
        }
        else_if = {
            limit = { scope:choose_various = yes }
            custom_tooltip = roe_sell_slaves_through_market_decision_choose_various_effect_tt
            show_as_tooltip = {
                add_gold = {
                    value = roe_min_slave_token_unit_value
                    multiply = roe_slave_price_per_unit_value
                }
            }
        }
        else_if = {
            limit = { scope:choose_all_slaves = yes }
            custom_tooltip = roe_sell_slaves_through_market_decision_choose_all_effect_tt
            show_as_tooltip = {
                add_gold = {
                    value = {
                        value = 0
                        add = var:roe_african_slave_token
                        add = var:roe_slavic_slave_token
                        add = var:roe_turkic_slave_token
                        add = var:roe_various_slave_token
                    }
                    multiply = roe_slave_price_per_unit_value
                }
            }
        }

        # どこに売るのか: realm
        if = {
            limit = {
                any_realm_county = {
                    roe_county_has_valid_slave_market_trigger = yes
                }
            }
            random_realm_county = {
                limit = {
                    roe_county_has_valid_slave_market_trigger = yes
                }
                save_scope_as = market_county
            }
        }
        # どこに売るのか: sub_realm
        if = {
            limit = {
                NOT = { exists = scope:market_county }
            }
            random_sub_realm_county = {
                limit = {
                    roe_county_has_valid_slave_market_trigger = yes
                }
                save_scope_as = market_county
            }
        }

        # 最終的に何unit売るのかの記録
        set_local_variable = {
            name = sell_token_unit_count
            value = 0
        }

        # 何unit売るのか
        if = { # 全部
            limit = { scope:choose_all_slaves = yes }
            change_local_variable = {
                name = sell_token_unit_count
                add = {
                    value = 0
                    add = var:roe_african_slave_token
                    add = var:roe_slavic_slave_token
                    add = var:roe_turkic_slave_token
                    add = var:roe_various_slave_token
                }
            }
        }
        else = { # 全部でないなら最小unit
            change_local_variable = {
                name = sell_token_unit_count
                add = roe_min_slave_token_unit_value
            }
        }

        # キャラクターからunitの取り出し
        # 全部
        if = {
            limit = { scope:choose_all_slaves = yes }
            roe_character_takeout_all_slave_token_effect = yes
        }
        # AFRICAN
        else_if = {
            limit = {
                scope:choose_african = yes
                exists = local_var:sell_token_unit_count
            }
            roe_character_takeout_slave_token_with_limit_effect = {
                MAX_AFRICAN_TOKEN = local_var:sell_token_unit_count
                MAX_SLAVIC_TOKEN = 0
                MAX_TURKIC_TOKEN = 0
                MAX_VARIOUS_TOKEN = 0
            }
        }
        # SLAVIC
        else_if = {
            limit = {
                scope:choose_slavic = yes
                exists = local_var:sell_token_unit_count
            }
            roe_character_takeout_slave_token_with_limit_effect = {
                MAX_AFRICAN_TOKEN = 0
                MAX_SLAVIC_TOKEN = local_var:sell_token_unit_count
                MAX_TURKIC_TOKEN = 0
                MAX_VARIOUS_TOKEN = 0
            }
        }
        # TURKIC
        else_if = {
            limit = {
                scope:choose_turkic = yes
                exists = local_var:sell_token_unit_count
            }
            roe_character_takeout_slave_token_with_limit_effect = {
                MAX_AFRICAN_TOKEN = 0
                MAX_SLAVIC_TOKEN = 0
                MAX_TURKIC_TOKEN = local_var:sell_token_unit_count
                MAX_VARIOUS_TOKEN = 0
            }
        }
        # VARIOUS
        else_if = {
            limit = {
                scope:choose_various = yes
                exists = local_var:sell_token_unit_count
            }
            roe_character_takeout_slave_token_with_limit_effect = {
                MAX_AFRICAN_TOKEN = 0
                MAX_SLAVIC_TOKEN = 0
                MAX_TURKIC_TOKEN = 0
                MAX_VARIOUS_TOKEN = local_var:sell_token_unit_count
            }
        }

        # marketに移譲
        if = {
            limit = {
                exists = scope:market_county
                exists = scope:RETURN_AFRICAN_SLAVE_TOKEN
                exists = scope:RETURN_SLAVIC_SLAVE_TOKEN
                exists = scope:RETURN_TURKIC_SLAVE_TOKEN
                exists = scope:RETURN_VARIOUS_SLAVE_TOKEN
            }
            scope:market_county = {
                roe_county_store_slave_token_effect = {
                    AFRICAN_SLAVE_TOKEN = scope:RETURN_AFRICAN_SLAVE_TOKEN
                    SLAVIC_SLAVE_TOKEN = scope:RETURN_SLAVIC_SLAVE_TOKEN
                    TURKIC_SLAVE_TOKEN = scope:RETURN_TURKIC_SLAVE_TOKEN
                    VARIOUS_SLAVE_TOKEN = scope:RETURN_VARIOUS_SLAVE_TOKEN
                }
            }
        }

        # 収入
        if = {
            limit = { exists = local_var:sell_token_unit_count }
            save_scope_value_as = {
                name = gain_gold
                value = {
                    value = local_var:sell_token_unit_count
                    multiply = roe_slave_price_per_unit_value
                }
            }
        }
        if = {
            limit = { exists = scope:gain_gold }
            add_gold = {
                value = scope:gain_gold
            }
        }

    }

    ai_check_interval = 6
    ai_potential = {
        faith.religion = religion:islam_religion
    }
    ai_will_do = {
        base = 100
    }
